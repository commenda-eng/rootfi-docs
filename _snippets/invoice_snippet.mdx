<ResponseField name="platform_id" type="string" required>
  The accounting platform's unique identifier for the invoice
</ResponseField>

<ResponseField name="contact_id" type="string" required>
  The unique identifier for the contact associated with the invoice
</ResponseField>

<ResponseField name="sales_order_ids" type="string[]" optional>
  An array of unique identifiers for the sales orders associated with the invoice
</ResponseField>

<ResponseField name="document_number" type="string" required>
  The unique reference number for the invoice
</ResponseField>

<ResponseField name="posted_date" type="date" required>
  The date the invoice was issued or posted
</ResponseField>

<ResponseField name="due_date" type="date" required>
  The date by which the invoice is due to be paid
</ResponseField>

<ResponseField name="currency_id" type="string">
  The accounting platform's unique identifier for the currency used in the invoice
</ResponseField>

<ResponseField name="currency" type="string">
  The ISO-4217 currency code for the currency used in the invoice
</ResponseField>

<ResponseField name="currency_rate" type="float">
  The exchange rate between the invoice currency and the base currency
</ResponseField>

<ResponseField name="status" type="enum">
  The status of the invoice. Example values can be: DRAFT, SENT, PAID, PARTIALLY_PAID, OVERDUE, etc.
</ResponseField>

<ResponseField name="total_amount" type="float">
  The total amount of the invoice
</ResponseField>

<ResponseField name="sub_total" type="float">
  The pre-tax total value of the invoice
</ResponseField>

<ResponseField name="total_discount" type="float">
  The total value of discounts applied to the invoice
</ResponseField>

<ResponseField name="tax_amount" type="float">
  The total tax amount levied on the invoice
</ResponseField>

<ResponseField name="amount_due" type="float">
  The remaining amount due for the invoice
</ResponseField>

<ResponseField name="memo" type="string">
  Additional notes or a memo regarding the invoice. This field is optional.
</ResponseField>

<ResponseField name="updated_at" type="date" required>
  The date and time at which the invoice information was last updated in the accounting platform
</ResponseField>
