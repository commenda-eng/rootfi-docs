```bash cURL
curl --request POST \
  --url https://api.rootfi.dev/accounting/tracking_categories/all \
  --header 'Content-Type: application/json' \
  --header 'api_key: <api-key>' \
  --data '{
  "take": 10,
  "skip": 0,
  "orderBy": {
    "name": "asc"
  },
  "filter": {
    "name": {
      "eq": "Test Name"
    }
  }
}'
```

```javascript NodeJS
import { RootFiClient } from "rootfi-api";

const rootfi = new RootFiClient({
  apiKey: "<api_key>",
});

await rootfi.accounting.trackingCategories.list({
  take: 10,
  skip: 0,
  filter: {
    name: { eq: "Test Name" },
  },
  orderBy: {
    name: "asc",
  },
});
```
